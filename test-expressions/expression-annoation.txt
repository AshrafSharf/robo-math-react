M = write(4, 14, "\frac{a}{b} + c = d")
T = subonly(M,"\frac{a}{b}")
cancel(T, "0", "u")

  Or with multiple items:
  M = write(4, 14, "x^2 + 2x + 1")
  T = subonly(M, "x")
  cancel(T, "", "u", "red")

  This will cancel all occurrences of "x" in the expression.

  More examples:
  // Cancel with replacement text
  M = write(4, 14, "\frac{5}{10}")
  T = subonly(M, "5")
  cancel(T, "1", "u")

  // Down diagonal cancel
  M = write(4, 14, "a + b - c")
  T = subonly(M, "b")
  cancel(T, "", "d")

  // X cancel
  M = write(4, 14, "wrong answer")
  T = subonly(M, "wrong")
  cancel(T, "", "x", "red")

// ============= Distance Marker Examples =============

// Basic distance marker with coordinates
G = g2d(0, 0, 16, 8, -10, 10, -10, 10, 1)
dm(G, 0, 0, 5, 0, "5 units")

// Distance marker using points
A = point(G, 2, 3)
B = point(G, 8, 5)
dm(G, A, B, "\\sqrt{40}")

// With text offset (text above line by 0.5 units)
dm(G, 0, -3, 6, -3, "d = 6", 0.5)

// With marker offset (parallel marker shifted by 1 unit)
dm(G, -5, 0, -5, 5, "5 units", 0.3, 1)
